import ballerina/http;
public type SupplierInfo2 record {
    readonly int supplierId = -1;
    string name;
    string shortName;
    string email; 
    string phoneNumber;
    Quote2[] quotes?;
};

public type Quote2 record {
    readonly int quoteId = -1;
    SupplierInfo2 supplier?;

    int maxQuantity;
    int period;
    string brandName;
    int unitPrice;
    int expiryDate;
    string regulatoryInfo;
};

service /quotes on new http:Listener(8081) {

    function init () returns error? {}

    resource function post .(@http:Payload Quote2 quote) returns Quote2|error {
        return quote;
    }
}
